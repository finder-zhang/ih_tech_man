//=======================================================================Short Terms

// Before we type some codes,we define some terminology below.

// RT : Short for router

// ZB : Short for zigbee

// CMD : Short for command , router send a job to zigbee coordinator

// RSP : Short for Response , zigbee reply router's CMD

// ASK : Zigbee coordinator send a job to router

// ANS : Short for Answer ,router reply zigbee's CMD

// CHG : Short for change

 

 

//==========================================================Working sequence overview

// Working sequence describe below:

// The RT send CMD_xxx to ZB,and then ZB reply RSP_xxx to RT

// If some device's status has change,ZB will send a ASK_xxx to RT,

// then RT reply ANS_xxx to ZB

 

 

//==========================================================Structure fields alignment

//Because of this structure is used in 8bit 8051 cpu and 32bit mips cpu,

//so we must align the fields with ONE BYTE.

 

 

//========================================================Structure fields description

// Structures field's description:

// All the structure have a field named uFromTo in the first position.

// It indicate that where this package come from and send to,

// we suppose it can reduce some error when we got the package.

 

// The second field in all structures is uCmd,

// see also the enum eRouterAndZigbeeCommands_t.

// When a communication is raise,CMD and RSP,ASK and ANS will have the

// same uCmd,in other words,we reply a job with the same uCmd.

 

// All RSP_xxx and ANS_xxx have a field named uResult,

// In many case,this field is just reserver for the future,

// because some CMD or ASK's result must be true

// We'd better check whether the uResult is true in our code,

// even in the case uResult must be true.

// If CMD or ASK is well done,we must set uResult = true,

// otherwise,let it false.

 

// We don't use any enum types in struct's field declaration,

// instand,we use U8 to declare it,and follow by a common to declare

// the field's type

// For example : U8 uDeviceType; // this type is eDeviceType_t

typedef vector<U8> CommandPkg;

 

//this is a function object use by find_if

class CmdPkgComparator

{

protected:

         U8             _uFromTo;

         U8             _uCmd;

public:

         CmdPkgComparator(U8 uFromTo)

         {

                   CmdPkgComparator(uFromTo,0);

         }

         CmdPkgComparator(U8 uFromTo,U8 uCmd)

         {

                   _uFromTo = uFromTo;

                   _uCmd = uCmd;

         }

         bool operator()(const CommandPkg* pCP)

         {

                   const CommandPkg& cp = *pCP;

                   if ( cp.size() < 2) {

                            return false;

                   }

                   if ( 0 != _uFromTo ) {         //if FromTo need to compare

                            if (cp[0] != _uFromTo) {

                                     return false;

                            }

                   }

                   if ( 0 != _uCmd ) {               //if Command need to compare

                            if (cp[1] != _uCmd) {

                                     return false;

                            }

                   }

                   return true;

         }

};
